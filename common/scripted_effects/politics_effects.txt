parliament_support_ruling_party = {
	if = {
        limit = {
            has_government = harmony
        }
        set_variable = {
            var = parliament_support
            value = party_popularity_100@harmony 
        }
        ROOT = {
            set_country_flag = law_in_hearing
        }
    }
    else_if = {
        limit = {
            has_government = supremacy

support_voracity_undecided_supremacy = {
    if = {
        limit = {
            NOT = {
                has_government = supremacy
                has_government = voracity
                has_government = neutral
            }
        }
        set_variable = { bi_partisan_support = party_popularity_100@supremacy }
        add_to_variable = { bi_partisan_support = party_popularity_100@neutral }
        add_to_variable = { bi_partisan_support = party_popularity_100@voracity }
        divide_variable = { bi_partisan_support = 2 }
        add_to_variable = { parliament_support = bi_partisan_support }
    }
    else_if = {
        limit = {
            has_government = supremacy
        }
        set_variable = { bi_partisan_support = party_popularity_100@neutral }
        add_to_variable = { bi_partisan_support = party_popularity_100@voracity }
        divide_variable = { bi_partisan_support = 2 }
        add_to_variable = { parliament_support = bi_partisan_support }
    }
    else_if = {
        limit = {
            has_government = neutral
        }
        set_variable = { bi_partisan_support = party_popularity_100@supremacy }
        add_to_variable = { bi_partisan_support = party_popularity_100@voracity }
        divide_variable = { bi_partisan_support = 2 }
        add_to_variable = { parliament_support = bi_partisan_support }
    }
    else = {

}
