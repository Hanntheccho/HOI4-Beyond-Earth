introduce_proposition = {
    if = {
        limit = {
            check_variable = { corruption_variable < 25 }
        }
        add_political_power = -25
    }
    else_if = {
        limit = {
            check_variable = { corruption_variable > 75 }
        }
        add_political_power = -75
    }
    else = {
        set_temp_variable = { bill_proposing_cost = corruption_variable }
        multiply_temp_variable = { bill_proposing_cost = -1 }
        add_political_power = bill_proposing_cost
    }
}

pierce_corruption_dont_check = {
    if = {
        limit = {
            check_variable = { corruption_min > 0 }
        }
        subtract_from_variable = { corruption_min = 10 }
    }
}

pierce_corruption_check = {
    if = {
        limit = {
            AND = {
                check_variable = { corruption_variable < corruption_min }
                check_variable = { corruption_min > 0 }
            }
        }
        subtract_from_variable = { corruption_min = 10 }
    }
}
