engineered_ecobalance = {
	if = {
		limit = {
			has_dynamic_modifier = {
				modifier = no_ecobalance
			}
		}
		remove_dynamic_modifier = { modifier = no_ecobalance }
		add_dynamic_modifier = { modifier = counter_ecobalance }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = minor_ecobalance
			}
		}
		remove_dynamic_modifier = { modifier = minor_ecobalance }
		add_dynamic_modifier = { modifier = counter_ecobalance }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = notable_ecobalance
			}
		}
		remove_dynamic_modifier = { modifier = notable_ecobalance }
		add_dynamic_modifier = { modifier = counter_ecobalance }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = major_ecobalance
			}
		}
		remove_dynamic_modifier = { modifier = major_ecobalance }
		add_dynamic_modifier = { modifier = counter_ecobalance }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = significant_ecobalance
			}
		}
		remove_dynamic_modifier = { modifier = significant_ecobalance }
		add_dynamic_modifier = { modifier = counter_ecobalance }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = terminal_ecobalance
			}
		}
		remove_dynamic_modifier = { modifier = terminal_ecobalance }
		add_dynamic_modifier = { modifier = counter_ecobalance }
	}
	else = {

	}
}

decrease_ecobalance = {
	if = {
		limit = {
			has_dynamic_modifier = {
				modifier = counter_ecobalance
			}
			ROOT = {
				has_country_flag = has_mind_tier3
			}
		}
		remove_dynamic_modifier = { modifier = counter_ecobalance }
		add_dynamic_modifier = { modifier = no_ecobalance }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = minor_ecobalance
			}

calculate_current_ecobalance = {
	clear_variable = ROOT.current_ecobalance
	every_owned_state = {
		limit = {
			has_dynamic_modifier = {
				modifier = notable_ecobalance
			}
		}
		set_variable = { THIS.state_ecobalance = 1 }
		add_to_variable = { ROOT.current_ecobalance = THIS.state_ecobalance }
	}
	every_owned_state = {
		limit = {
			has_dynamic_modifier = {
				modifier = major_ecobalance
			}
		}
		set_variable = { THIS.state_ecobalance = 2 }
		add_to_variable = { ROOT.current_ecobalance = THIS.state_ecobalance }
	}
	every_owned_state = {
		limit = {
			has_dynamic_modifier = {
				modifier = significant_ecobalance
			}
		}
		set_variable = { THIS.state_ecobalance = 3 }
		add_to_variable = { ROOT.current_ecobalance = THIS.state_ecobalance }
	}
	every_owned_state = {
		limit = {
			has_dynamic_modifier = {
				modifier = terminal_ecobalance
			}
		}
		set_variable = { THIS.state_ecobalance = 4 }
		add_to_variable = { ROOT.current_ecobalance = THIS.state_ecobalance }
	}
}
