engineered_devastation = {
	if = {
		limit = {
			has_dynamic_modifier = {
				modifier = no_devastation
			}
		}
		remove_dynamic_modifier = { modifier = no_devastation }
		add_dynamic_modifier = { modifier = counter_devastation }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = minor_devastation
			}
		}
		remove_dynamic_modifier = { modifier = minor_devastation }
		add_dynamic_modifier = { modifier = counter_devastation }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = notable_devastation
			}
		}
		remove_dynamic_modifier = { modifier = notable_devastation }
		add_dynamic_modifier = { modifier = counter_devastation }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = major_devastation
			}
		}
		remove_dynamic_modifier = { modifier = major_devastation }
		add_dynamic_modifier = { modifier = counter_devastation }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = significant_devastation
			}
		}
		remove_dynamic_modifier = { modifier = significant_devastation }
		add_dynamic_modifier = { modifier = counter_devastation }
	}
	else = {

	}
}

decrease_devastation = {
	if = {
		limit = {
			has_dynamic_modifier = {
				modifier = counter_devastation
			}
			ROOT = {
				has_country_flag = has_mind_tier3
			}
		}
		remove_dynamic_modifier = { modifier = counter_devastation }
		add_dynamic_modifier = { modifier = no_devastation }
	}
	else_if = {
		limit = {
			has_dynamic_modifier = {
				modifier = minor_devastation
			}
			ROOT = {
				has_country_flag = has_mind_tier3
			}
		}

		add_dynamic_modifier = { modifier = no_devastation }

	}
}

calculate_current_devastation = {
	clear_variable = ROOT.current_devastation
	every_owned_state = {
		limit = {
			has_dynamic_modifier = {
				modifier = notable_devastation
			}
		}
		set_variable = { THIS.state_devastation = 1 }
		add_to_variable = { ROOT.current_devastation = THIS.state_devastation }
	}
	every_owned_state = {
		limit = {
			has_dynamic_modifier = {
				modifier = major_devastation
			}
		}
		set_variable = { THIS.state_devastation = 2 }
		add_to_variable = { ROOT.current_devastation = THIS.state_devastation }
	}
	every_owned_state = {
		limit = {
			has_dynamic_modifier = {
				modifier = significant_devastation
			}
		}
		set_variable = { THIS.state_devastation = 3 }
		add_to_variable = { ROOT.current_devastation = THIS.state_devastation }
	}
	every_owned_state = {
		limit = {
			has_dynamic_modifier = {
				modifier = terminal_devastation
			}
		}
		set_variable = { THIS.state_devastation = 4 }
		add_to_variable = { ROOT.current_devastation = THIS.state_devastation }
	}
}
