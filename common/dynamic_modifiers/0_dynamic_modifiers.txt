#BUILDING RESOURCES

building_resources_used = {
	enable = { always = yes }
	
	#icon = GFX_modifiers_sabotaged_resource
	
	state_resource_cost_energy = building_resources_used_energy
 	state_resource_cost_polymers = building_resources_used_polymers
 	state_resource_cost_lithium = building_resources_used_lithium
 	state_resource_cost_electronics = building_resources_used_electronics
 	state_resource_cost_superalloys = building_resources_used_superalloys
 	state_resource_cost_superconductors = building_resources_used_superconductors
}

too_many_building_resources_used = {
	enable = { 
		always = yes
	}
	remove_trigger = {
		NOT = { 
			OR = {
				check_variable = {
					resource@energy < 0
				}
				check_variable = {
					resource@polymers < 0
				}
				check_variable = {
					resource@electronics < 0
				}
				check_variable = {
					resource@superalloys < 0
				}
				check_variable = {
					resource@superconductors < 0
				}
				check_variable = {
					resource@lithium < 0
				}
			}
		}
	}
	
	#icon = GFX_modifiers_sabotaged_resource
	
	production_speed_buildings_factor = too_many_building_resources_used_var
}
