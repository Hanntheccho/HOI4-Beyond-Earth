 	state_resource_cost_aluminium = building_resources_used_aluminium
 	state_resource_cost_rubber = building_resources_used_rubber
 	state_resource_cost_tungsten = building_resources_used_tungsten
 	state_resource_cost_steel = building_resources_used_steel
 	state_resource_cost_chromium = building_resources_used_chromium
}

too_many_building_resources_used = {
	enable = { 
		always = yes
	}
	remove_trigger = {
		NOT = { 
			OR = {
				check_variable = {
					resource@oil < 0
				}
				check_variable = {
					resource@aluminium < 0
				}
				check_variable = {
					resource@tungsten < 0
				}
				check_variable = {
					resource@steel < 0
				}
				check_variable = {
					resource@chromium < 0
				}
				check_variable = {
					resource@rubber < 0
				}
			}
		}
	}
	
	#icon = GFX_modifiers_sabotaged_resource
	
	production_speed_buildings_factor = too_many_building_resources_used_var
}
