	on_declare_war = {
		effect = {
			every_country = {
				if = {
					limit = {
						any_guaranteed_country = {
							tag = FROM
						}
					}
					declare_war_on = {
						target = ROOT
						type = puppet_wargoal_focus
					}
				}
			}
		}
	}
	#
	on_puppet = {
		effect = {
			ROOT = {
				every_owned_state = {
					limit = {
						OR = {
							is_claimed_by = FROM
							is_core_of = FROM
						}
					}
					transfer_state_to = FROM
				}
			}
		}
	}

			}
		}
	}
