on_actions = {

	on_startup = {
		effect = {
			every_country = {
				limit = { 
                    has_cultural_power = yes
                }
                set_variable = { cultural_power = 0 }
                set_variable = { cultural_power_max = 2000 }
				set_variable = { cultural_power_rate_stable = 1 }
				set_variable = { cultural_power_rate_mobilizing = 0.75 }
				set_variable = { cultural_power_rate_mobilized = 0.5 }
				set_variable = { cultural_power_rate_radicalized = 0.25 }
				set_variable = { cultural_power_rate_rabid = 0 }
            }
		}
	}
    
    on_daily = {
		effect = {
			if = {
				limit = { 
                    has_idea = stable_culture
                }
                
                add_to_variable = { cultural_power = cultural_power_rate_stable }
				
                clamp_variable = { var = cultural_power min = 0 max = cultural_power_max }
			}
			if = {
				limit = { 
                    has_idea = mobilizing_culture
                }
                
                add_to_variable = { cultural_power = cultural_power_rate_mobilizing }
				
                clamp_variable = { var = cultural_power min = 0 max = cultural_power_max }
			}
			if = {
				limit = { 
                    has_idea = mobilized_culture
                }
                
                add_to_variable = { cultural_power = cultural_power_rate_mobilized }
				
                clamp_variable = { var = cultural_power min = 0 max = cultural_power_max }
			}
			if = {
				limit = { 
                    has_idea = radicalized_culture
                }
                
                add_to_variable = { cultural_power = cultural_power_rate_radicalized }
				
                clamp_variable = { var = cultural_power min = 0 max = cultural_power_max }
			}
			if = {
				limit = { 
                    has_idea = rabid_culture
                }
                
                add_to_variable = { cultural_power = cultural_power_rate_rabid }
				
                clamp_variable = { var = cultural_power min = 0 max = cultural_power_max }
			}
        }
    }
}
