on_actions = {
	on_startup = {
		effect = {
            set_variable = { global.industrial_power_rate_agrarian = 0.2 }
            set_variable = { global.industrial_power_rate_developing = 1 }
            set_variable = { global.industrial_power_rate_developed = 0.8 }
            set_variable = { global.industrial_power_rate_regional = 0.5 }
            set_variable = { global.industrial_power_rate_continental = 0.2 }
            set_variable = { global.industrial_power_rate_global = 0.1 }
            every_country = {
                set_variable = {
                    industrial_power_gain = num_of_civilian_factories
                }
                divide_variable = {
                    industrial_power_gain = 10
                }
                if = {
                    limit = { 
                        has_idea = agrarian_stage
                    }                
                    multiply_variable = { industrial_power_gain = global.industrial_power_rate_agrarian }
                }
                else_if = {
                    limit = { 
                        has_idea = developing_stage
                    }                
                    multiply_variable = { industrial_power_gain = global.industrial_power_rate_developing }
                }
                else_if = {
                    limit = { 
                        has_idea = developed_stage
                    }                
                    multiply_variable = { industrial_power_gain = global.industrial_power_rate_developed }
                }
                else_if = {
                    limit = { 
                        has_idea = regional_stage
                    }                
                    multiply_variable = { industrial_power_gain = global.industrial_power_rate_regional }
                }
                else_if = {
                    limit = { 
                        has_idea = continental_stage
                    }                
                    multiply_variable = { industrial_power_gain = global.industrial_power_rate_continental }
                }
                else = {
                    multiply_variable = { industrial_power_gain = global.industrial_power_rate_global }
                } 
            }
		}
	}
###
    on_daily = {
		effect = {
            clear_variable = industrial_power_gain
            set_variable = {
                industrial_power_gain = num_of_civilian_factories
            }
            divide_variable = {
                industrial_power_gain = 10
            }
			if = {
				limit = { 
                    has_idea = agrarian_stage
                }                
                multiply_variable = { industrial_power_gain = global.industrial_power_rate_agrarian }
			}
			else_if = {
				limit = { 
                    has_idea = developing_stage
                }                
                multiply_variable = { industrial_power_gain = global.industrial_power_rate_developing }
			}
			else_if = {
				limit = { 
                    has_idea = developed_stage
                }                
                multiply_variable = { industrial_power_gain = global.industrial_power_rate_developed }
			}
			else_if = {
				limit = { 
                    has_idea = regional_stage
                }                
                multiply_variable = { industrial_power_gain = global.industrial_power_rate_regional }
			}
			else_if = {
				limit = { 
                    has_idea = continental_stage
                }                
                multiply_variable = { industrial_power_gain = global.industrial_power_rate_continental }
			}
			else = {
				multiply_variable = { industrial_power_gain = global.industrial_power_rate_global }
			}
            add_to_variable = { industrial_power = industrial_power_gain }
            clamp_variable = { var = industrial_power min = 0 max = 2000 }
        }
    }
}
