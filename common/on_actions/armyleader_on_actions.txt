on_actions = {
    on_unit_leader_level_up = { 
		effect = {
        	if = {
        	    limit = { 
        	        skill > 7
					NOT = { has_character_flag = this_dude_is_human }
        	    }
        	    set_leader_portrait = GFX_portrait_ai_command_lvl_8
        	}
        	else_if = {
        	    limit = { 
        	        skill > 6
					NOT = { has_character_flag = this_dude_is_human }
        	    }
        	    set_leader_portrait = GFX_portrait_ai_command_lvl_7
        	}
        	else_if = {
        	    limit = { 
        	        skill > 5
					NOT = { has_character_flag = this_dude_is_human }
        	    }
        	    set_leader_portrait = GFX_portrait_ai_command_lvl_6
        	}
        	else_if = {
        	    limit = { 
        	        skill > 4
					NOT = { has_character_flag = this_dude_is_human }
        	    }
        	    set_leader_portrait = GFX_portrait_ai_command_lvl_5
        	}
        	else_if = {
        	    limit = { 
        	        skill > 3
					NOT = { has_character_flag = this_dude_is_human }
        	    }
        	    set_leader_portrait = GFX_portrait_ai_command_lvl_4
        	}
        	else_if = {
        	    limit = { 
        	        skill > 2
					NOT = { has_character_flag = this_dude_is_human }
        	    }
        	    set_leader_portrait = GFX_portrait_ai_command_lvl_3
        	}
        	else_if = {
        	    limit = { 
        	        skill > 1
					NOT = { has_character_flag = this_dude_is_human }
        	    }
        	    set_leader_portrait = GFX_portrait_ai_command_lvl_2
        	}
        	else_if = {
        	    limit = { 
        	        skill > 0
					NOT = { has_character_flag = this_dude_is_human }
        	    }
        	    set_leader_portrait = GFX_portrait_ai_command_lvl_1
        	}
        	else = {
			
        	}
		}
    }
}
