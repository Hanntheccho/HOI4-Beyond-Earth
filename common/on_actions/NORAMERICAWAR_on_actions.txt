on_actions = {
	###
	on_startup = {
		effect = {
			every_country = {
				limit = {
					OR = {
						is_nau_tag = yes
						tag = PNM
						tag = ARC
					}
				}
				add_ideas = NAU_dont_join_me
			}
			set_variable = { global.napgain_var = 0.5 }
			ONE = {
				set_variable = { nau_legitimacy = 4 }
				set_variable = { arc_legitimacy = 3 }
			}
			DOS = {
				set_variable = { nau_legitimacy = 4 }
				set_variable = { arc_legitimacy = 6 }
			}
			TRE = {
				set_variable = { nau_legitimacy = 1 }
				set_variable = { arc_legitimacy = 4 }
			}
			CUA = {
				set_variable = { nau_legitimacy = 4 }
				set_variable = { arc_legitimacy = 1 }
			}
			CIN = {
				set_variable = { nau_legitimacy = 5 }
				set_variable = { arc_legitimacy = 3 }
			}
			SEI = {
				set_variable = { nau_legitimacy = 3 }
				set_variable = { arc_legitimacy = 5 }
			}
			SIT = {
				set_variable = { nau_legitimacy = 4 }
				set_variable = { arc_legitimacy = 4 }
			}
			OCH = {
				set_variable = { nau_legitimacy = 3 }
				set_variable = { arc_legitimacy = 5 }
			}
		}
	}
	on_weekly_ADU = {
		effect = {
			if = {
				limit = {
					NOT = {
						has_global_flag = noramerica_war_happened
					}
				}
				add_to_variable = {
					var = global.noramericawarprogress_var
					value = global.napgain_var
				}
				if = {
					limit = {
						check_variable = {
							var = global.noramericawarprogress_var
							value = 100
							compare = greater_than_or_equals
						}
					}
					every_country = {
						limit = {
							has_idea = NAU_dont_join_me
						}
						remove_ideas = NAU_dont_join_me
					}
					every_country = {
						limit = {
							is_nau_tag = yes
							check_variable = {
								nau_legitimacy < arc_legitimacy
							}
						}
						leave_faction = yes
						ARC = {
							add_to_faction = PREV
						}
					}
					every_country = {
						limit = {
							OR = {
								is_in_faction_with = ARC
								tag = ARC
							}
						}
						every_country = {
							limit = {
								OR = {
									is_in_faction_with = PNM
									tag = PNM
								}
							}
							THIS = {
								declare_war_on = {
									target = PREV.PREV
									type = annex_puppet_wargoal
								}
							}
						}
					}
					set_global_flag = noramerica_war_happened
				}
			}
			else = {
				#
			}
		}
	}
}
