on_actions = {
	###
	on_startup = {
		effect = {
			set_variable = { global.napgain_var = 0.5 }
			ONE = {
				set_variable = { nau_legitimacy = 4 }
				set_variable = { arc_legitimacy = 3 }
			}
			DOS = {
				set_variable = { nau_legitimacy = 4 }
				set_variable = { arc_legitimacy = 6 }
			}
			TRE = {
				set_variable = { nau_legitimacy = 1 }
				set_variable = { arc_legitimacy = 4 }
			}
			CUA = {
				set_variable = { nau_legitimacy = 4 }
				set_variable = { arc_legitimacy = 1 }
			}
			CIN = {
				set_variable = { nau_legitimacy = 5 }
				set_variable = { arc_legitimacy = 3 }
			}
			SEI = {
				set_variable = { nau_legitimacy = 3 }
				set_variable = { arc_legitimacy = 5 }
			}
			SIT = {
				set_variable = { nau_legitimacy = 4 }
				set_variable = { arc_legitimacy = 4 }
			}
			OCH = {
				set_variable = { nau_legitimacy = 3 }
				set_variable = { arc_legitimacy = 5 }
			}
		}
	}
	on_weekly_ADU = {
		effect = {
			add_to_variable = {
				var = global.noramericawarprogress_var
				value = global.napgain_var
			}
			if = {
				limit = {
					check_variable = {
						var = global.noramericawarprogress_var
						value = 100
    					compare = greater_than_or_equals
					}
				}
				every_country = {
					limit = {
						check_variable = {
							nau_legitimacy < arc_legitimacy
						}
					}
					leave_faction = yes
					ARC = {
						add_to_faction = PREV
					}
				}
				PNM = {
					declare_war_on = {
						target = ARC
						type = annex_puppet_wargoal
					}
				}
			}
		}
	}
	###
	on_puppet = {
		effect = {
			every_country = {
				limit = {
					is_puppet_of = PNM
					OR = {
						tag = OCH
						tag = SIT
						tag = SEI
						tag = CIN
						tag = CUA
						tag = TRE
						tag = DOS
						tag = ONE
					}
					NOT = {
						has_idea = NAU_central_authority_mid
					}
				}
				add_ideas = NAU_central_authority_mid
			}
		}
	}
}
