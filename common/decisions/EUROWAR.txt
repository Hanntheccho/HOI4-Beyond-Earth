						}
					}
					NOT = {
						OR = {
							is_puppet_of = ROOT
							has_war_with = ROOT
							is_in_faction_with = ROOT
						}
					}
				}
			}
		}
		fire_only_once = no
		days_remove = 30
		war_with_target_on_remove = yes
		remove_effect = {
			declare_war_on = {
				target = FROM.OWNER
				type = annex_puppet_wargoal
			}
		}
		ai_will_do = {
			factor = 0
			modifier = {
				NOT = {
					any_enemy_country = {
						is_neighbor_of = ROOT
						any_controlled_state = {
							is_in_home_area = yes
						}
					}
				}
				add = 100
			}
		}
	}
