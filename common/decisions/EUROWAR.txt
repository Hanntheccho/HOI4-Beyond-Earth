		icon = GFX_decision_war

		visible = {

		}
		target_trigger = {
			FROM = {
				capital_scope = {
					is_on_continent = europe
				}
				OR = {
					is_neighbor_of = ROOT
					ROOT = {
						any_allied_country = {
							is_neighbor_of = FROM
						}
					}
				}
				NOT = {
					OR = {
						is_puppet_of = ROOT
						has_war_with = ROOT
						is_in_faction_with = ROOT
					}						
				}
			}
		}
		fire_only_once = no
		complete_effect = {
			declare_war_on = {
				target = FROM
				type = annex_puppet_wargoal
			}
		}
		ai_will_do = {
			factor = 0
			modifier = {
				NOT = {
					any_enemy_country = {
						is_neighbor_of = ROOT
						any_controlled_state = {
							is_in_home_area = yes
						}
					}
				}				
				add = 100
			}
		}
	}
