	TZN_africa_destiny = {
		icon = GFX_decision_eaf
		state_target = yes
		on_map_mode = map_only
		available = {
			is_puppet = no
		}
		visible = {
			has_global_flag = TZN_united_africa
		}
		target_trigger = {
			FROM = {
				is_capital = yes
				OWNER = {
					capital_scope = {
						is_on_continent = africa
					}
					OR = {
						is_neighbor_of = ROOT
						ROOT = {
							any_country = {
								OR = {
									is_puppet_of = PREV
									is_in_faction_with = PREV
								}
								is_neighbor_of = FROM.OWNER
							}
						}
					}
					NOT = {
						OR = {
							is_puppet_of = ROOT
							has_war_with = ROOT
							#is_in_faction_with = ROOT
						}					
					}
				}
			}
		}
		fire_only_once = no
		days_remove = 30
		complete_effect = {
			if = {
				limit = {
					FROM.OWNER = {
						is_in_faction_with = ROOT
					}
				}
				ROOT = {
					remove_from_faction = FROM.OWNER
				}
			}
			hidden_effect = {
				set_country_flag = {
					flag = going_to_war
					days = 30
					value = 1
				}
			}
		}
		war_with_target_on_remove = yes
		remove_effect = {
			declare_war_on = {
				target = FROM.OWNER
				type = annex_puppet_wargoal
			}
		}
		ai_will_do = {
			factor = 0
			modifier = {
				has_war = no
				add = 100
			}
			modifier = {
				has_country_flag = going_to_war
				add = -1000
			}
			modifier = {
				FROM.OWNER = {
					is_in_faction_with = ROOT
				}
				add = -1000
			}
		}
	}
