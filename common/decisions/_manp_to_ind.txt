state_decisions = {
	man_to_civ_targ = {
		visible = {
			#
		}		
		available = {
			custom_trigger_tooltip = {
				check_variable = {
					var = industrial_power
					value = 30
					compare = greater_than_or_equals
				}
				tooltip = state_decisions_cost
			}
			FROM = {
				NOT = {
					has_state_flag = already_doing_something_here
				}
			}
		}

		state_target = yes
		
		target_root_trigger = {
		}

		target_trigger = {
			FROM = {
				is_fully_controlled_by = ROOT
				is_owned_by = ROOT
				is_core_of = ROOT
				free_building_slots = {
					building = industrial_complex
					size > 0
					include_locked = no
				}
				if = {
					limit = {
						ROOT = {
							any_controlled_state = {
								infrastructure > 4
								free_building_slots = {
									building = industrial_complex
									size > 0
									include_locked = no
								}
							}
						}						
					}
					infrastructure > 4
				}
				else_if = {
					limit = {
						ROOT = {
							any_controlled_state = {
								infrastructure > 3
								free_building_slots = {
									building = industrial_complex
									size > 0
									include_locked = no
								}
							}
						}						
					}
					infrastructure > 3
				}
				else_if = {
					limit = {
						ROOT = {
							any_controlled_state = {
								infrastructure > 2
								free_building_slots = {
									building = industrial_complex
									size > 0
									include_locked = no
								}
							}
						}						
					}
					infrastructure > 2
				}
				else = {
					#
				}
			}
		}
		
		on_map_mode = map_only
		icon = GFX_decision_industrial

		days_remove = 60
		ai_will_do = {
			base = 1
			modifier = {
				check_variable = {
					industrial_power > 200
				}
				add = 10
			}
			modifier = {
				check_variable = {
					industrial_power_gain < 0
				}
				add = 10
			}
		}		
		complete_effect = {
			add_to_variable = {
				industrial_power = -30
			}
			FROM = {
				set_state_flag = {
					flag = already_doing_something_here
					days = 60
					value = 1
				}
			}
		}			
		remove_effect = {
			FROM = {
				add_building_construction = {
					type = industrial_complex
					level = 1
					instant_build = yes
				}
			}
		}
	}
