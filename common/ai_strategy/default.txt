default_building_production_civilian = {
	allowed = {
		always = yes
	}
	enable = {
		has_war = no
		any_owned_state = {
			OR = {
				free_building_slots = {
					building = industrial_complex
					size > 0
					include_locked = no
				}
				free_building_slots = {
					building = spaceship_factory
					size > 0
					include_locked = no
				}
				free_building_slots = {
					building = arms_factory
					size > 0
					include_locked = no
				}
			}
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = building_target
		id = industrial_complex
		value = 30
	}
	ai_strategy = {
		type = building_target
		id = spaceship_factory
		value = 5
	}
	ai_strategy = {
		type = building_target
		id = arms_factory
		value = 10
	}
}
