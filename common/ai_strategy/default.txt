division_limiter = {
	enable = {
		OR = {
			AND = {
        		check_variable = { num_divisions > division_limit }
				NOT = { surrender_progress > 0.7 }
			}
			is_puppet = yes
		}
	}
	
	abort_when_not_enabled = yes

	ai_strategy = {
		type = build_army
		id = artillery
		value = -1000
	}

	ai_strategy = {
		type = build_army
		id = mobile
		value = -1000

		value = -1000
	}
