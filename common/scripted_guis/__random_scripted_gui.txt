	diplo_options_gui = {
		context_type = selected_country_context
		parent_window_token = selected_country_view_diplomacy
		
		window_name = "diplo_options"

        visible = {
			always = yes
        }

		effects = {
            war_button_click = {
				hidden_effect = {
					if = {
						limit = {
							ROOT = {
								has_idea = digital_democracy
							}
						}
						ROOT = {
							add_political_power = -200
						}						
					}
					else_if = {
						limit = {
							ROOT = {
								has_idea = autocratic_authority
							}
						}
						ROOT = {
							add_political_power = -100
						}
					}
					else = {
						ROOT = {
							add_political_power = -150
						}						
					}
				}
				if = {
					limit = {
						THIS = {
							is_puppet_of = ROOT
						}
					}
					ROOT = {
						end_puppet = PREV
					}
				}
				if = {
					limit = {
						THIS = {
							is_in_faction_with = ROOT
						}
					}
					if = {
						limit = {
							ROOT = {
								is_faction_leader = yes
							}
							
						}
						THIS = {
							leave_faction = yes
						}
					}
					else = {
						ROOT = {
							leave_faction = yes
						}
					}
				}
				ROOT = {
					declare_war_on = {
						type = annex_puppet_wargoal
						target = PREV
					}
				}				
			}
		}

		triggers = {
            war_button_click_enabled = {
				THIS = {
					NOT = {
						#is_puppet_of = ROOT
						tag = ROOT
					}
				}
				if = {
					limit = {
						ROOT = {
							has_idea = digital_democracy
						}
					}
					ROOT = {
						has_political_power > 199
					}
				}
				else_if = {
					limit = {
						ROOT = {
							has_idea = autocratic_authority
						}
					}
					ROOT = {
						has_political_power > 99
					}
				}
				else = {
					ROOT = {
						has_political_power > 149
					}
				}
			}
        }

        ai_enabled = {
            always = no
        }
    }

                    limit = {
                        has_country_flag = international_moment_flag

                    clr_country_flag = international_moment_flag
                }

                    set_country_flag = international_moment_flag
                }
            }
            economic_button_click = {
