scripted_gui = {
	scavenging_decisions_scripted_gui = {
		context_type = decision_category
		
		window_name = "scavenging_decisions_scripted_gui"
		
		visible = {
            tag = FRA
			#has_country_flag = ruinous_trait
		}

		effects = {
			scavenging_decisions_scripted_gui_additional_scavengers_click = {
				set_variable = {has_sent_additional_scavengers = 1}
				add_to_variable = {monthly_scavenger_rate = 10}
				subtract_from_variable = {industrial_power = 6}
			}
			scavenging_decisions_scripted_gui_locate_foreign_sponsors_click = {
				set_variable = {has_located_foreign_sponsors = 1}
				add_to_variable = {international_aid_available = 1}
				add_political_power = -50
			}
			scavenging_decisions_scripted_gui_take_out_loans_click = {
				set_variable = {has_taken_out_loans_scavengers = 1}
				country_event = {id=scav.1}
				subtract_from_variable = {industrial_power = 10}
			}
		}

		triggers = {
			scavenging_decisions_scripted_gui_additional_scavengers_click_enabled = {
				AND = {
					check_variable = {
						var = areas_left_unsearched
						value = 10
						compare = greater_than_or_equals
					}
					check_variable = {
						var = industrial_power
						value = 6
						compare = greater_than_or_equals
					}
					has_war = no
				}
			}
			scavenging_decisions_scripted_gui_locate_foreign_sponsors_click_enabled = {
				AND = {
					check_variable = {
						var = international_aid_available
						value = 10
						compare = less_than_or_equals
					}
					has_political_power > 50
				}
			}
			scavenging_decisions_scripted_gui_take_out_loans_click_enabled = {
				AND = {
					NOT = {
						has_idea = anarchy_style
					}
					has_war = no
					check_variable = {
						var = industrial_power
						value = 10
						compare = greater_than_or_equals
					}
				}
			}
		}
	}
}
