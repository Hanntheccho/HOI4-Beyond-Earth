add_namespace = superbug_event

news_event = {
	id = superbug_event.1
	title = superbug_event.1.t
	desc = superbug_event.1.d
	picture = GFX_superbug_event.1
	
	major = yes  
	
	trigger = { 
		has_global_flag = vax_prepremoment
		date > 2200.01.05
	}
	
	mean_time_to_happen = { days = 5 }
	
	option = { 
		name = superbug_event_option.1.a 
	}
	immediate = {
		hidden_effect = {
			GER = {
				every_owned_state = {
					limit = { state_population > 15000000 }
					add_dynamic_modifier = { modifier = superbug_high }
				}
				every_owned_state = {
					limit = { state_population > 5000000  state_population < 14999999 }
					add_dynamic_modifier = { modifier = superbug_medium }
				}
				every_owned_state = {
					limit = { state_population > 0  state_population < 4999999 }
					add_dynamic_modifier = { modifier = superbug_low }
				}
			}
			clr_global_flag = vax_prepremoment
			set_global_flag = vax_premoment
		}
	}
}

				limit = { state_population > 5000000  state_population < 14999999 NOT = { has_dynamic_modifier = { modifier = superbug_medium } } }

			}

				limit = { state_population > 0  state_population < 4999999 NOT = { has_dynamic_modifier = { modifier = superbug_low } } }
				add_dynamic_modifier = { modifier = superbug_low }
			}
			clr_global_flag = vax_premoment
			set_global_flag = vax_momento
		}
	}
}
