country_event = {
	id = centralasia_event.10
	title = centralasia_event.10.t
	desc = centralasia_event.10.d
	picture = GFX_unknown_event

	fire_only_once = yes
	
	is_triggered_only = yes
	
	option = { 
		name = centralasia_event_option.10.a
		if = {
			limit = {
				check_variable = { undecided_ticker > harmony_ticker }
				check_variable = { undecided_ticker > divinity_ticker }
				check_variable = { undecided_ticker > supremacist_ticker }
			}
			country_event = {
				id = centralasia_event.11
			}
		}
		else = {
			limit = {
				check_variable = { harmony_ticker > undecided_ticker }
				check_variable = { harmony_ticker > divinity_ticker }
				check_variable = { harmony_ticker > supremacist_ticker }
			}
			country_event = {
				id = centralasia_event.12
			}
		}
		else = {
			limit = {
				check_variable = { divinity_ticker > undecided_ticker }
				check_variable = { divinity_ticker > harmony_ticker }
				check_variable = { divinity_ticker > supremacist_ticker }
			}
			country_event = {
				id = centralasia_event.13
			}
		}
		else = {
			limit = {
				check_variable = { supremacist_ticker > undecided_ticker }
				check_variable = { supremacist_ticker > harmony_ticker }
				check_variable = { supremacist_ticker > divinity_ticker }
			}
			country_event = {
				id = centralasia_event.14
			}
		}
	}
}
