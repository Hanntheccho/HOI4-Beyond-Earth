news_event = {
	id = newyear_event.2
	title = newyear_event.2.t
	desc = newyear_event.2.d
	picture = GFX_newyear_event.2
	
	major = yes  
	
	fire_only_once = yes
	
	trigger = { is_ai = no }
	
	mean_time_to_happen = { days = 5 }
	
	option = { 
		name = newyear_event_option.2.a 
	}
	immediate = {
		hidden_effect = {
			every_country = { add_ideas = superbugs }
			every_state = {
				limit = { state_population > 15000000 }
				add_dynamic_modifier = { modifier = superbug_high }
			}
			every_state = {
				limit = { state_population > 5000000  state_population < 14999999 }
				add_dynamic_modifier = { modifier = superbug_medium }
			}
			every_state = {
				limit = { state_population > 1000000  state_population < 4999999 }
				add_dynamic_modifier = { modifier = superbug_low }
			}
		}
	}
}
