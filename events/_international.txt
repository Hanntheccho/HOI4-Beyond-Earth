	major = yes

	trigger = {
		check_variable = {
			var = global.threat_debug
			value = 100
			compare = greater_than_or_equals
		}

	}

	immediate = {
		hidden_effect = {
			set_global_flag = civilization_collapses_flag
			every_state = {
				increase_ecobalance = yes
			}
		}
		custom_effect_tooltip = civilization_collapses_event_tt
	}

	mean_time_to_happen = {
		days = 5
	}

	option = {
		name = international_news_event.7.a
	}
}

	id = international_news_event.13
	title = international_news_event.13.t
	desc = international_news_event.13.d
	#picture = GFX_international_news_event.13

	is_triggered_only = yes

	option = {
		name = international_news_event.13.a #do internationalize
		custom_effect_tooltip = international_news_event.13.a_tt
		hidden_effect = {
			set_country_flag = spaceship_internationalized_flag
		}
		ai_chance = {
			factor = 1000
			modifier = {
				OR = {
					has_idea = autocratic_authority
					has_idea = unstable_leadership
					has_idea = closed_economy
					if = {
						limit = {
							has_idea = oligarchic_government
						}
						has_war = yes
					}
				}
				add = -1000
			}
		}
	}

	option = {
		name = international_news_event.13.b #do NOT internationalize
		custom_effect_tooltip = international_news_event.13.b_tt
		hidden_effect = {
			if = {
				limit = {
					NOT = {
						has_global_flag = final_war_flag
					}
				}
				set_global_flag = final_war_flag
				news_event = {
					id = international_news_event.14
					days = 20
				}
			}
			set_country_flag = not_spaceship_internationalized_flag
		}
		ai_chance = {
			factor = 10
		}
	}
}
