	id = cin_event.2
	title = cin_event.2.t
	desc = cin_event.2.d
	picture = GFX_unknown_event_picture

	is_triggered_only = yes

	major = yes
	
	option = { 
		name = cin_event_option.2.a
		trigger = {
			capital_scope = {
				is_on_continent = north_america
			}
			check_variable = {
				arc_legitimacy > nau_legitimacy
			}
		}
	}
	option = { 
		name = cin_event_option.2.b
		trigger = {
			capital_scope = {
				is_on_continent = north_america
			}
			NOT = {
				check_variable = {
					arc_legitimacy > nau_legitimacy
				}
			}
		}
	}
	option = { 
		name = cin_event_option.2.c
		trigger = {
			NOT = {
				capital_scope = {
					is_on_continent = north_america
				}
			}
		}		
	}
}
